pipeline {
    agent any

    stages {

        stage('Clone Repository') {
            steps {
                git branch: 'main', url: 'https://github.com/Balarajuvelpula/project-aws-098.git'
            }
        }

        stage('List Project Files') {
            steps {
                sh '''
                echo "Project Files:"
                ls -la
                '''
            }
        }

        stage('Run Script') {
            steps {
                sh '''
                echo "Running project script"
                chmod +x *.sh || true
                bash *.sh || true
                '''
            }
        }

        stage('Deploy Stage') {
            steps {
                sh '''
                echo "Deployment completed successfully"
                '''
            }
        }

    }
}
